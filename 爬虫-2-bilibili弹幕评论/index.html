def hack_wrid(params):
    for item in ['w_rid','wts']:
        if item in params:params.pop(item)

    with open('1.评论逆向.js','r',encoding="utf-8")as f:
        js_code = f.read()
    ctx = execjs.compile(js_code).call('res',params)
    # 破解w_rid（wts）是时间戳
    params['w_rid'] = ctx['w_rid']
    params['wts'] = ctx['wts']
    return params



def get_comments(comments,d):
    if d == {}:d = {
        'message': [],
        'like': [],
        'ctime': [],
        'uname': [],
        'sex': [],
        'location': [],
        'current_level': [],
        'vipStatus': [],
        'vipType': [],
        'rcount': [],
    }
    for i in range(len(comments)):
        # 评论
        d['message'].append(comments[i]['content']['message'])
        # 点赞
        d['like'].append(comments[i]['like'])
        # 时间
        d['ctime'].append(comments[i]['ctime'])
        # 昵称
        d['uname'].append(comments[i]['member']['uname'])
        # 性别
        d['sex'].append(comments[i]['member']['sex'])
        # ip地址
        if 'location' not in comments[i]['reply_control']:
            d['location'].append('')
        else:
            d['location'].append(comments[i]['reply_control']['location'])
        # 等级
        d['current_level'].append(comments[i]['member']['level_info']['current_level'])
        # 会员状态
        d['vipStatus'].append(comments[i]['member']['vip']['vipStatus'])
        # 会员等级
        d['vipType'].append(comments[i]['member']['vip']['vipType'])
        # 回复数量
        d['rcount'].append(comments[i]['rcount'])
    return d
